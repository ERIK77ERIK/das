JAVASCRIPT - Զանգվածներ(Arrays) - Ցուցակ



JS 8

1.Ցուցակ(array) մենք ստեղծում ենք հետևյալ կերպ []

2.Ցուցակը(array-ը) արժեք է և նույն ձև ոնց որ կարող ենք արժեք(թիվ, տեքստ կամ բուլյան արժեք) վերագրել փոփոխականի մեջ նույն ձև մենք կարող ենք ցուցակը վերագրել փոփոխականի մեջ

3.Ցուցակի մեջ մենք կարող ենք դնել արժեքներ(որը կլինի ցուցակի անդամները), ցուցակի մեջի արժեքները մենք կարող ենք կարդալ նրանց index(ցուցանիշ)-ով այսինքն այն թվով թե ինքը որ երորդն է, ցուցակի անդամները մենք սկսում ենք հաշվել 0-ից այսինքն առաջի անդամը կլինի 0-րդը, երկրորդ անդամը կլինի 1-րդը(առաջի անդամը),երրորդ անդամը կլինի 2-րդը(երկրորդ անդամը)

4.
Ցուցակի մեջի արժեքները մենք կարող ենք վերագրել ինչ-որ փոփոխականի
let x = [44, 33, 22, 11];
let y = x[0];
alert(y);
կտպի 44

Ցուցակի մեջի արժեքներին մենք կարող ենք ուրիշ արժեքներ վերագրել
let x = [44, 33, 22, 11];
x[1] = 8;
alert(x[1]);
կտպի 8

Կարող ենք ստեղծել դատարկ ցուցակ և հետո մեջը վերագրել արժեքներ
let x = [];
x[4] = 1;
x[5] = 2;
alert(x);
alert(x[4]);
alert(x.length);
կտպի (դատրակություն այսինքն y-ին վերագրած զանգվածի մեջի պատկերը) հետո 1 հետո 6

5.Ցուցակը ունի length(երկարություն) հասկացողությունը,length-ը արժեք է որը ցույց է տալիս ցուցակի մեջ եղած արժեքների քանակը
ցուցակի մեջի արժեքների քանակը հավասար է վերջի անդամի index-ին գումարած մեկ, քանի որ index-ը 0-ից ենք սկսել հաշվել
Ցուցակի արժեքների քանակը կարդում ենք հետևյալ կերպ գրվում է ( փոփոխականի անունը(որին վերագրած է ցուցակը) կետ length )
let x = [4, 56, 76];
alert(x.length);
կտպի 3

2.
let a = ["Poghos", "Petros", "Martiros", "Aramis", "Nika"];
alert(a);

let ai = 0;
let vi = a.length-1;

while(ai < a.length/2){
let d=a[ai];
a[ai]=a[vi];
a[vi]=d;
ai=ai + 1;
vi=vi - 1;
}
alert(a);

կտպի Poghos,Petros,Martiros,Aramis,Nika հետո Nika, Aramis, Martiros, Petros, Poghos

let y = [];
let n = 0;

while(n < b.length){
y[n]=b[n] * 2;
n=n + 1;
}
alert(b);
