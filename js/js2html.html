JAVASCRIPT - Փոփոխականներ(let)



JS 1

1.Փոփոխականները ստեղծում ենք let տեքստով,գրում ենք let և մեր արժեքի անունը(փոփոխականը), կարող ենք վերագրել ինչ-որ արժեք անմիջապես կամ ստեղծել վերագրել ապագայում
2.( = ) -ը ծրագրաորման մեջ նշանակում է վերագրել, այսինքն աջ կողմը վարագրվում է ձախ կողմի մեջ

let x=7,y=5;
alert( x + y + "բարև" + x + y);

3.javascript-ը թույլ չի տալիս երկու փոփոխական ստեղծել նույն անունով բայց կարող ենք նույն փոփոխականին տված արժեքը փոխել
եթե նույն փոփոխականին տված արժեքը փոխում ենք ապա alert ֆունկցիան ընդունում է անմիջապես վերևի տված արժեքը
նաև կարող ենք տարբեր փոփոխականի նույն արժեքը տալ

let a=4,b=5;
a=7;
a=1;
alert(a);
alert(a+b);

alert(a+b);
