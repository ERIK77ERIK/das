1. \n-ը ստեղծում է նոր տող և նշանակում է new line(նոր տող), գրվում է ""-ի մեջ որպեսզի չհամարվի \ նշանը բաժանում n-ն փոփոխականի անուն

2. Զանգված(array)-ը ունի իր վրա ֆունկցիաներ(հրամաններ) որոնցից մեկը կոչվում է join, join նշանակում է ( միացնել )
join-ը վերցնում է բոլոր անդամները ցուցակ(array)-ի, իրանց վերածում է տեքստի և կպցնում է իրար մի ուրիշ տեքստով
join ֆունկցիան գրվում է փոփոխականի անունը(որին վերագրած է ցուցակը) կետ join և վերջում փակագիծներ որի մեջ մենք կարող ենք գրել այն արժեքը որով ցանկանում ենք կպցնել ցուցակի անդամներին
let x = [5, 2, 6];
alert(x.join());
փակագիծների մեջ եթե դատարկ թողնենք join-ը կվերցնի ցուցակի անդամներին եվ կկպցնի ստորակետով
այսինքն կտպի 5,2,6 որը կտպվեր նույն ձև նաև ուղղակի ցուցակը տպելուց alert(x);

let x = [5, 2, 6];
alert(x.join(""));
կտպի 526

let x = [5, 2, 6];
alert(x.join(" "));
կտպի 5 2 6

3.Սկզբից մենք պետք է ներկայացնենք թե ոնցա խաղը ընթանալու
Հետո հերթականությունը լավ պետք է բացատրենք թե ինչիա սկզբից տախտակը տպում, ոնցա լինում որ index-ի տվյալներով քարը հայտնվումա տախտակի վրա և այլն
Մարդու լոգիկան զարգացնելու համար պետքա նախ առաջինը որ իր միտքը հանգիստ լինի այսինքն եթե մարդը մտածումա թե ոնց գրի այդ կոդը արդեն իսկ նա չի կարող մտածել թե ոնց գրի այդ լոգիկան
Կոդը գրելուց լավ պետք է պատկերացնենք թե մեր գրած կոդը ոնց է տպվելու այսինքն եթե մենք լավ գիտենք կոդի իմաստը թե ինչա անում բայց չենք պատկերացնում թե ոնց է տպվելու կամ հակառակը ուզում ենք տպել մեր պատկերացրած ձև բայց չգիտենք ոնց գրենք այդ կոդը որ այդպես տպվի այդ դեպքում մենք չենք կարող կենտրոնանալ
